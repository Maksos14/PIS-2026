# job_offer_controller.py

class JobOfferController:
    """
    REST-контроллер для работы с откликами на вакансии.
    """

    def __init__(self, job_offer_service):
        self.job_offer_service = job_offer_service

    def create_offer(self, request):
        """
        Обрабатывает POST-запрос на создание отклика.
        """
        command = request.to_command()
        offer_id = self.job_offer_service.create_offer(command)
        return {"offer_id": offer_id, "status": "created"}

    def get_offer(self, offer_id: str):
        """
        Обрабатывает GET-запрос на получение отклика.
        """
        offer = self.job_offer_service.get_offer(offer_id)
        if offer is None:
            return {"error": "Not found"}, 404
        return self._format_response(offer)

# Часть 6. Dependency Injection (Конфигурация зависимостей)

DependencyContainer (DI‑контейнер):

python
# dependency_injection.py

from application.service.job_offer_service import JobOfferService
from infrastructure.adapter.out.in_memory_job_offer_repository import InMemoryJobOfferRepository
from infrastructure.adapter.out.console_notification_service import ConsoleNotificationService

class DependencyContainer:
    """
    DI-контейнер: связывает входящие и исходящие порты
    с их реализациями в инфраструктурном слое.
    """

    def __init__(self):
        # Исходящие адаптеры
        self.job_offer_repository = InMemoryJobOfferRepository()
        self.notification_service = ConsoleNotificationService()

        # Application Service
        self.job_offer_service = JobOfferService(
            repository=self.job_offer_repository,
            notification_service=self.notification_service
        )

    def get_job_offer_service(self):
        return self.job_offer_service
Как работает DI:

DI‑контейнер создаёт реализации исходящих портов (JobOfferRepository, NotificationService).

Эти реализации передаются в JobOfferService через конструктор.

Контроллеры и другие входящие адаптеры получают готовый сервис через методы контейнера.

Таким образом, application‑слой не создаёт зависимости сам, а получает их извне, что соответствует принципу Dependency Inversion.

Что получилось хорошо
Удалось чётко разделить архитектуру на три независимых слоя: Domain, Application и Infrastructure. Доменная модель (JobOffer, User, Project) получилась чистой и не зависит от инфраструктуры. С какими трудностями столкнулись
Возникли сложности с пониманием роли входящих и исходящих портов, особенно когда у интерфейса пока только одна реализация. DI‑контейнер потребовал аккуратной настройки зависимостей, чтобы не нарушить принцип инверсии зависимостей.

Что узнали нового
Познакомился с концепцией Hexagonal Architecture и принципами Ports & Adapters. Усвоил важность Dependency Inversion и того, что доменный слой должен быть полностью независим от инфраструктуры. Понял, как порты позволяют легко подменять адаптеры (например, InMemory → PostgreSQL). Разобрался, как DI упрощает тестирование и расширение системы.
